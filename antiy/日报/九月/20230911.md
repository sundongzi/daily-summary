```vue
 <!-- 该组件的样式基于element -->
<template>
  <div
    class="el-progress el-progress--dashboard"
    role="progressbar"
    :aria-valuenow="percentage"
    aria-valuemin="0"
    aria-valuemax="100"
  >
    <div class="el-progress-circle" :style="{height: width + 'px', width: width + 'px'}">
      <svg viewBox="0 0 100 100">
        <path
          class="el-progress-circle__track"
          :d="trackPath"
          :stroke="defineBackColor"
          :stroke-width="relativeStrokeWidth"
          fill="none"
          :style="trailPathStyle"></path>
        <path
          class="el-progress-circle__path"
          :d="trackPath"
          :stroke="stroke"
          fill="none"
          :stroke-linecap="strokeLinecap"
          :stroke-width="percentage ? relativeStrokeWidth : 0"
          :style="circlePathStyle"></path>
      </svg>
    </div>
  </div>
</template>
<script>
  const RATE = 0.5
  export default {
    name: 'SysProgress',
    props: {
      percentage: {
        type: Number,
        default: 0,
        required: true,
        validator: val => val >= 0 && val <= 100
      },
      strokeWidth: {
        type: Number,
        default: 8
      },
      strokeLinecap: {
        type: String,
        default: 'butt'
      },
      width: {
        type: Number,
        default: 126
      },
      successColor: {
        type: String,
        default: '#14CA64'
      },
      errorColor: {
        type: String,
        default: '#FF5959'
      },
      criticalValue: {
        type: Number,
        default: 50
      },
      color: {
        type: String,
        default: ''
      },
      defineBackColor: {
        type: String,
        default: '#d9d9d9'
      }
    },
    computed: {
      relativeStrokeWidth() {
        return (this.strokeWidth / this.width * 100).toFixed(1);
      },
      radius() {
        return parseInt(50 - parseFloat(this.relativeStrokeWidth) / 2, 10)
      },
      trackPath() {
        const radius = this.radius;
        return `
          M 50 50
          m 0 ${radius}
          a ${radius} ${radius} 0 1 1 0 -${radius * 2}
          a ${radius} ${radius} 0 1 1 0 ${radius * 2}
          `;
      },
      perimeter() {
        return 2 * Math.PI * this.radius;
      },
      strokeDashoffset() {
        const offset = -1 * this.perimeter * (1 - RATE) / 2;
        return `${offset}px`;
      },
      trailPathStyle() {
        return {
          strokeDasharray: `${(this.perimeter * RATE)}px, ${this.perimeter}px`,
          strokeDashoffset: this.strokeDashoffset
        };
      },
      circlePathStyle() {
        return {
          strokeDasharray: `${this.perimeter * RATE * (this.percentage / 100) }px, ${this.perimeter}px`,
          strokeDashoffset: this.strokeDashoffset,
          transition: 'stroke-dasharray 0.6s ease 0s, stroke 0.6s ease'
        };
      },
      stroke() {
        if (this.color) return this.color
        return this.percentage > this.criticalValue ? this.errorColor : this.successColor;
      }
    }
  };
</script>
```
