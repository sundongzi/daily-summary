1. 折线图、柱状图、饼图组件封装

   ```vue
   // 单主机详情饼图配置
   computed: {
     pieOption () {
         return {
           legend: {
             top: 'center',
             right: '20px',
             height: '80%',
             orient: 'vertical',
             formatter: (name) => {
               const data = this.pieData ?? []
               for (let i = 0; i < data.length; i++) {
                 if (name === data[i].name) {
                   return '{name|' + name + '}{value|' + data[i].value + '}'
                 }
               }
             },
             textStyle: {
               rich: {
                 name: {
                   fontSize: 14,
                   fontWeight: 400,
                   width: 100,
                   height: 20,
                   padding: [0, 0, 0, 5],
                   color: 'rgba(0, 0, 0, 0.85)'
                 },
   
                 value: {
                   fontSize: 14,
                   fontWeight: 500,
                   height: 20,
                   width: 50,
                   align: 'left',
                   color: 'rgba(0, 0, 0, 0.65)'
                 }
               }
             }
           },
           series: [
             {
               label: {
                 show: true,
                 position: 'center',
                 formatter: '{b_style|{d}%}\n{c_style|{b}率}',
                 rich: {
                   b_style: {
                     fontSize: 22,
                     color: '#202226',
                     fontWeight: 'bold'
                   },
                   c_style: {
                     color: '#7B808C',
                     fontSize: 20,
                     padding: [5, 0, 5, 0]
                   }
                 }
               },
               emphasis: {
                 disabled: true
               }
   
             }
           ]
         }
       }
   }
   ```
